cmake_minimum_required(VERSION 3.10 FATAL_ERROR)

project(lib-sysinfo VERSION 0.1.0)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} /MT")
set(CMAKE_CXX_FLAGS_MINSIZEREL "${CMAKE_CXX_FLAGS_RELEASE} /MT")
set(CMAKE_CXX_FLAGS_RELWITHDEBINFO "${CMAKE_CXX_FLAGS_RELEASE} /MT")
set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} /MTd")
set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} /FORCE:Multiple")


aux_source_directory(src ALL_SRC)

add_library(${PROJECT_NAME} STATIC ${ALL_SRC})
target_include_directories(${PROJECT_NAME} PUBLIC include)
set_target_properties(${PROJECT_NAME} PROPERTIES LINKER_LANGUAGE CXX)
target_compile_definitions(${PROJECT_NAME} PRIVATE LIBSYSINFO_EXPORT)

add_executable(${PROJECT_NAME}-test test/main.cpp)
target_include_directories(${PROJECT_NAME}-test PUBLIC include)
target_link_libraries(${PROJECT_NAME}-test PRIVATE ${PROJECT_NAME})
set_target_properties(${PROJECT_NAME}-test PROPERTIES LINKER_LANGUAGE CXX)
